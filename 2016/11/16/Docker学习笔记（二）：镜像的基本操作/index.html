在镜像内创建一个新文件:
[wb@localhost ~]$ sudo docker run -ti ubuntu:14.04 /bin/bash
root@dc46bcb6858a:/# touch test
root@dc46bcb6858a:/# ls
bin boot dev etc home lib lib64 media mnt opt proc root run sbin srv sys test tmp usr var
此时容器ID为dc46bcb6858a，保存为新镜像：
[wb@localhost ~]$ sudo docker commit -m “Add a new file” dc46bcb6858a test
sha256:9790710e4c85127f960d32f64b1fb9e76944f1922df70de27b18bb61f472c30e
[wb@localhost ~]$ sudo docker images
REPOSITORY TAG IMAGE ID CREATED SIZE
test latest 9790710e4c85 13 seconds ago 187.9 MB

[wb@localhost ~]$ sudo cat ubuntu-12.04-x86-minimal.tar.gz | sudo docker import - ubuntu:12.04
sha256:574af84d86e9dc97ffa47875f31173d0095c5ac917ea4d35441c60bf2cc31baf

[wb@localhost ~]$ sudo docker images
REPOSITORY TAG IMAGE ID CREATED SIZE
ubuntu 12.04 574af84d86e9 15 seconds ago 145.8 MB

[wb@localhost ~]$ sudo docker save -o ubuntu_1404.tar ubuntu:14.04

[wb@localhost ~]$ sudo docker load –input ubuntu_1404.tar
或者
[wb@localhost ~]$ sudo docker load < ubuntu_1404.tar
